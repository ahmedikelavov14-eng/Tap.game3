import asyncio import sqlite3 from aiogram import Bot, Dispatcher, types, F from aiogram.filters import Command from aiogram.utils.keyboard import InlineKeyboardBuilder # Вставь сюда свой токен TOKEN = "ТВОЙ_ТОКЕН_ОТ_BOTFATHER" bot = Bot(token=TOKEN) dp = Dispatcher() # Работа с базой данных def init_db(): conn = sqlite3.connect("tapper_db.sqlite") cursor = conn.cursor() cursor.execute(""" CREATE TABLE IF NOT EXISTS users ( user_id INTEGER PRIMARY KEY, clicks INTEGER DEFAULT 0 ) """) conn.commit() conn.close() def get_clicks(user_id): conn = sqlite3.connect("tapper_db.sqlite") cursor = conn.cursor() cursor.execute("SELECT clicks FROM users WHERE user_id = ?", (user_id,)) result = cursor.fetchone() conn.close() return result[0] if result else 0 def add_click(user_id): conn = sqlite3.connect("tapper_db.sqlite") cursor = conn.cursor() cursor.execute(""" INSERT INTO users (user_id, clicks) VALUES (?, 1) ON CONFLICT(user_id) DO UPDATE SET clicks = clicks + 1 """, (user_id,)) conn.commit() conn.close() # Клавиатура с кнопкой-тапкой def get_game_kb(count): builder = InlineKeyboardBuilder() builder.button(text=f"👟 ТАП! (Всего: {count})", callback_data="tap") return builder.as_markup() @dp.message(Command("start")) async def start_cmd(message: types.Message): user_id = message.from_user.id count = get_clicks(user_id) await message.answer( f"Привет! Это игра-тапалка.\nНажимай на кнопку ниже!", reply_markup=get_game_kb(count) ) @dp.callback_query(F.data == "tap") async def tap_handler(callback: types.Callback_query): user_id = callback.from_user.id add_click(user_id) new_count = get_clicks(user_id) # Обновляем сообщение, чтобы изменить счетчик на кнопке try: await callback.message.edit_reply_markup( reply_markup=get_game_kb(new_count) ) except: # Если кликают слишком быстро, Telegram может выдать ошибку — игнорируем её pass await callback.answer() async def main(): init_db() await dp.start_polling(bot) if __name__ == "__main__": asyncio.run(main())
